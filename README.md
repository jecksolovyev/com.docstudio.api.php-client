    $quickSendRequest->setSubject('Test envelope')->setMessage('Test envelope message');
    $sender = new QuickSendRecipientDTO();
    $recipients = [];
    $sender->setRecipient($mailboxUuid)->setSigner(true)->setEInkSignature(true);
